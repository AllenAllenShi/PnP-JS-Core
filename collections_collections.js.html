"use strict";
/**
 * Generic dictionary
 */
class Dictionary {
    /**
     * Creates a new instance of the Dictionary<T> class
     *
     * @constructor
     */
    constructor(keys = [], values = []) {
        this.keys = keys;
        this.values = values;
    }
    /**
     * Gets a value from the collection using the specified key
     *
     * @param key The key whose value we want to return, returns null if the key does not exist
     */
    get(key) {
        let index = this.keys.indexOf(key);
        if (index < 0) {
            return null;
        }
        return this.values[index];
    }
    /**
     * Adds the supplied key and value to the dictionary
     *
     * @param key The key to add
     * @param o The value to add
     */
    add(key, o) {
        let index = this.keys.indexOf(key);
        if (index > -1) {
            this.values[index] = o;
        }
        else {
            this.keys.push(key);
            this.values.push(o);
        }
    }
    /**
     * Merges the supplied typed hash into this dictionary instance. Existing values are updated and new ones are created as appropriate.
     */
    merge(source) {
        if ("getKeys" in source) {
            let sourceAsDictionary = source;
            sourceAsDictionary.getKeys().map(key => {
                this.add(key, sourceAsDictionary.get(key));
            });
        }
        else {
            let sourceAsHash = source;
            for (let key in sourceAsHash) {
                if (sourceAsHash.hasOwnProperty(key)) {
                    this.add(key, sourceAsHash[key]);
                }
            }
        }
    }
    /**
     * Removes a value from the dictionary
     *
     * @param key The key of the key/value pair to remove. Returns null if the key was not found.
     */
    remove(key) {
        let index = this.keys.indexOf(key);
        if (index < 0) {
            return null;
        }
        let val = this.values[index];
        this.keys.splice(index, 1);
        this.values.splice(index, 1);
        return val;
    }
    /**
     * Returns all the keys currently in the dictionary as an array
     */
    getKeys() {
        return this.keys;
    }
    /**
     * Returns all the values currently in the dictionary as an array
     */
    getValues() {
        return this.values;
    }
    /**
     * Clears the current dictionary
     */
    clear() {
        this.keys = [];
        this.values = [];
    }
    /**
     * Gets a count of the items currently in the dictionary
     */
    count() {
        return this.keys.length;
    }
}
exports.Dictionary = Dictionary;
