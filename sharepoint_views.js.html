/**
 * Describes the views available in the current context
 *
 */
class Views extends queryable_1.QueryableCollection {
    /**
     * Creates a new instance of the Views class
     *
     * @param baseUrl The url or Queryable which forms the parent of this fields collection
     */
    constructor(baseUrl) {
        super(baseUrl, "views");
    }
    /**
     * Gets a view by guid id
     *
     * @param id The GUID id of the view
     */
    getById(id) {
        let v = new View(this);
        v.concat(`('${id}')`);
        return v;
    }
    /**
     * Gets a view by title (case-sensitive)
     *
     * @param title The case-sensitive title of the view
     */
    getByTitle(title) {
        return new View(this, `getByTitle('${title}')`);
    }
    /**
     * Adds a new view to the collection
     *
     * @param title The new views's title
     * @param personalView True if this is a personal view, otherwise false, default = false
     * @param additionalSettings Will be passed as part of the view creation body
     */
    /*tslint:disable max-line-length */
    add(title, personalView = false, additionalSettings = {}) {
        let postBody = JSON.stringify(util_1.Util.extend({
            "__metadata": { "type": "SP.View" },
            "Title": title,
            "PersonalView": personalView,
        }, additionalSettings));
        return this.postAs({ body: postBody }).then((data) => {
            return {
                data: data,
                view: this.getById(data.Id),
            };
        });
    }
}
exports.Views = Views;

     * Updates this view intance with the supplied properties
     *
     * @param properties A plain object hash of values to update for the view
     */

        let postBody = JSON.stringify(util_1.Util.extend({
            "__metadata": { "type": "SP.View" },
        }, properties));
        return this.post({
            body: postBody,
            headers: {
                "X-HTTP-Method": "MERGE",
            },
        }).then((data) => {
            return {
                data: data,
                view: this,
            };
        });
    }
    /**
     * Delete this view

     */
    delete() {
        return this.post({
            headers: {
                "X-HTTP-Method": "DELETE",
            },
        });
    }

     * Returns the list view as HTML.

    renderAsHtml() {
        let q = new queryable_1.Queryable(this, "renderashtml");
